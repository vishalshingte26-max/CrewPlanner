import streamlit as st
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from sklearn.linear_model import LinearRegression

st.title("Airline Crew Planning & Rostering Tool")

st.sidebar.header("Operational Inputs")

# Airline parameters
aircraft = st.sidebar.number_input("Number of Aircraft", 1, 1000, 430)
flights_per_aircraft = st.sidebar.slider("Flights per Aircraft per Day", 1, 10, 6)
avg_flight_hours = st.sidebar.slider("Average Flight Hours", 1.0, 5.0, 2.0)

# Aircraft configuration
seats = st.sidebar.number_input("Seats per Aircraft", 50, 300, 180)
load_factor = st.sidebar.slider("Expected Load Factor", 0.5, 1.0, 0.85)

# Crew rules
pilot_max_hours = st.sidebar.slider("Pilot Max Flying Hours (DGCA)", 6, 12, 8)
buffer = st.sidebar.slider("Reserve Crew Buffer %", 0.05, 0.25, 0.15)

# Training pipeline
pilot_training_months = st.sidebar.slider("Pilot Training Months", 1, 12, 5)
cabin_training_months = st.sidebar.slider("Cabin Crew Training Months", 1, 6, 2)

st.header("Historical Demand Input")

months = st.number_input("Number of Historical Months", 3, 24, 12)

data = []
for i in range(months):
    value = st.number_input(f"Passengers Month {i+1}", 100000, 20000000, 1000000)
    data.append(value)

df = pd.DataFrame({
    "month": np.arange(len(data)).reshape(-1,1).flatten(),
    "passengers": data
})

# Forecast using linear regression
X = df["month"].values.reshape(-1,1)
y = df["passengers"].values

model = LinearRegression()
model.fit(X,y)

future_month = len(data)
forecast_passengers = model.predict([[future_month]])[0]

st.subheader("Demand Forecast")
st.write(f"Forecast passengers next month: **{int(forecast_passengers):,}**")

# Flights required
flights_required = forecast_passengers / (seats * load_factor)

st.subheader("Flights Required")

st.write(f"Estimated flights needed: **{int(flights_required):,} per month**")

# Convert to daily flights
daily_flights = flights_required / 30

# Flight hours
total_flight_hours = daily_flights * avg_flight_hours

# Crew calculations
pilot_shifts = total_flight_hours / pilot_max_hours
pilots_required = pilot_shifts * (1 + buffer)

cabin_required = daily_flights * 4
cabin_required = cabin_required * (1 + buffer)

ground_staff = daily_flights * 8
ground_staff = ground_staff * (1 + buffer)

st.header("Optimal Crew Requirement")

crew_df = pd.DataFrame({
    "Role": ["Pilots", "Cabin Crew", "Ground Staff"],
    "Required": [int(pilots_required), int(cabin_required), int(ground_staff)]
})

st.table(crew_df)

st.header("Training Pipeline Recommendation")

pilot_training_start = pilot_training_months
cabin_training_start = cabin_training_months

st.write(f"Pilot training should begin **{pilot_training_start} months before demand increase**.")
st.write(f"Cabin crew training should begin **{cabin_training_start} months before demand increase**.")

st.header("Crew Utilization Risk")

available_hours = pilots_required * pilot_max_hours
utilization = total_flight_hours / available_hours

if utilization > 0.9:
    st.error("High delay risk: Crew utilization too high")
elif utilization > 0.8:
    st.warning("Moderate risk")
else:
    st.success("Operationally stable")

st.header("Demand Visualization")

fig, ax = plt.subplots()

ax.plot(df["month"], df["passengers"], label="Historical")
ax.scatter(future_month, forecast_passengers, color="red", label="Forecast")

ax.set_xlabel("Month")
ax.set_ylabel("Passengers")
ax.legend()

st.pyplot(fig)

st.header("Consultant Recommendation")

st.write(
"""
Maintain crew utilization between **70–85%**.
If utilization rises above 90%, additional reserve crew or earlier training should be planned.
"""
)
